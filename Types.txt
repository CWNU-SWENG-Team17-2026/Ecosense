4. 'src/stores/useNoiseStore.ts'
import { create } from 'zustand';
import { persist } from 'zustand/middleware';
import { NoiseState, Spike, SessionType } from '../types/store';

interface NoiseStore extends NoiseState {
  setCurrentDb: (db: number) => void;
  setBackgroundDb: (db: number) => void;
  addSpike: (spike: Spike) => void;
  setSessionId: (id: string | null) => void;
  setIsSpiking: (isSpiking: boolean) => void;
  setIsMeasuring: (isMeasuring: boolean) => void;

  startNewSession: (type: SessionType) => string;
  endCurrentSession: () => void;
  clearCurrentSpikes: () => void;
}

export const useNoiseStore = create<NoiseStore>()(
  persist(
    (set, get) => ({
      currentDb: -Infinity,
      backgroundDb: -45,
      spikes: [],
      isMeasuring: false,
      sessionId: null,
      isSpiking: false,
      sessionStartTime: null,           // ← 초기값 필수!

      setCurrentDb: (db) => set({ currentDb: db }),
      setBackgroundDb: (db) => set({ backgroundDb: db }),
      addSpike: (spike) => set((state) => ({ spikes: [...state.spikes, spike] })),
      setSessionId: (id) => set({ sessionId: id }),
      setIsSpiking: (isSpiking) => set({ isSpiking }),
      setIsMeasuring: (isMeasuring) => set({ isMeasuring }),

      startNewSession: (type: SessionType) => {
        const id = crypto.randomUUID();
        const startTime = new Date().toISOString();

        set({
          sessionId: id,
          sessionStartTime: startTime,
          spikes: [],
          isSpiking: false,
        });

        console.log(`🆕 새 ${type} 세션 시작: ${id} (${startTime})`);
        return id;
      },

      endCurrentSession: () => {
        const { sessionId } = get();
        if (sessionId) console.log(`🏁 세션 종료: ${sessionId}`);

        set({ 
          sessionId: null, 
          sessionStartTime: null,   // ← 같이 초기화 (중요!)
          isSpiking: false 
        });
      },

      clearCurrentSpikes: () => set({ spikes: [] }),
    }),
    {
      name: 'noise-storage',
      partialize: (state) => ({
        spikes: state.spikes,
        sessionId: state.sessionId,
        sessionStartTime: state.sessionStartTime,   // ← persist에도 추가!
      }),
    }
  )
);

5. useSync
// src/hooks/useSync.ts
import { useCallback, useEffect } from 'react';
import { useAuthStore } from '../stores/useAuthStore';
import { useNoiseStore } from '../stores/useNoiseStore';
import api from '../services/api';
import { SyncUploadRequest } from '../types/api';
import { getSpikesBySession, saveSpikeWithBlob } from '../services/indexedDB';

export const useSync = () => {
  const { isLoggedIn } = useAuthStore();
  const { 
    sessionId, 
    spikes, 
    sessionStartTime, 
    startNewSession, 
    endCurrentSession 
  } = useNoiseStore();

  /** 현재 세션 업로드 */
  const uploadToServer = useCallback(async () => {
    if (!isLoggedIn || !sessionId) return;

    try {
      const sessionSpikes = await getSpikesBySession(sessionId);

      const payload: SyncUploadRequest = {
        sessions: [
          {
            id: sessionId,
            type: 'SLEEP',
            started_at: sessionStartTime || new Date().toISOString(),
            ended_at: new Date().toISOString(),
          },
        ],
        spikes: sessionSpikes.map((s) => ({
          session_id: s.session_id,
          detected_at: s.detected_at,
          db_level: s.db_level,
          duration_sec: s.duration_sec,
        })),
      };

      const res = await api.post('/sync/upload', payload);
      console.log(`✅ 서버 업로드 성공! ${res.data.synced_spikes}개 스파이크`);
    } catch (err) {
      console.error('❌ 서버 업로드 실패:', err);
    }
  }, [isLoggedIn, sessionId, sessionStartTime]);

  /** 서버 다운로드 + merge */
  const downloadFromServer = useCallback(async () => {
    if (!isLoggedIn) return;

    try {
      const res = await api.get('/sync/download?days=30');
      const serverSpikes = res.data?.spikes || [];

      let mergedCount = 0;
      for (const spike of serverSpikes) {
        await saveSpikeWithBlob(spike, new Blob()); // blob은 클라이언트 전용
        mergedCount++;
      }

      console.log(`✅ 서버 다운로드 + merge 완료! ${mergedCount}개 스파이크`);
    } catch (err) {
      console.error('❌ 서버 다운로드 실패:', err);
    }
  }, [isLoggedIn]);

  /** 자동 업로드 (debounce) */
  useEffect(() => {
    if (!isLoggedIn || !sessionId || spikes.length === 0) return;

    const timeout = setTimeout(uploadToServer, 2000);
    return () => clearTimeout(timeout);
  }, [isLoggedIn, sessionId, spikes.length, uploadToServer]);

  /** 로그인 직후 다운로드 */
  useEffect(() => {
    if (isLoggedIn) downloadFromServer();
  }, [isLoggedIn, downloadFromServer]);

  return { uploadToServer, downloadFromServer };
};
